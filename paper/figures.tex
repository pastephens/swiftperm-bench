\begin{figure}[ht]
\centering
\begin{tikzpicture}
\begin{axis}[
  ybar,
  ymode=log,
  log origin=infty,
  bar width=6pt,
  width=0.92\textwidth,
  height=7cm,
  enlarge x limits=0.12,
  ylabel={Throughput (perm/s, log scale)},
  ylabel style={font=\small},
  xtick=data,
  xticklabels from table={\throughputdata}{label},
  xticklabel style={font=\small},
  yticklabel style={font=\small},
  ytick={100, 1000, 10000, 100000, 1000000, 10000000},
  yticklabels={100, 1k, 10k, 100k, 1M, 10M},
  ymin=50, ymax=30000000,
  grid=major,
  grid style={dotted, gray!40},
  legend style={
    at={(0.98,0.98)},
    anchor=north east,
    font=\small,
    draw=gray!50,
    cells={anchor=west},
  },
  legend columns=1,
  title style={font=\small\bfseries},
  title={Moran's I permutation throughput by implementation and dataset size},
]

\addplot[fill=clgray,   draw=clgray!70!black]
  table[x expr=\coordindex, y=numpy] {\throughputdata};

\addplot[fill=clteal,   draw=clteal!70!black]
  table[x expr=\coordindex, y=numba] {\throughputdata};

\addplot[fill=clblue,   draw=clblue!70!black]
  table[x expr=\coordindex, y=metal] {\throughputdata};

\legend{NumPy (baseline), Numba parallel, Swift+Metal}

\end{axis}
\end{tikzpicture}
\caption{
  Permutation throughput (99,999 permutations) across implementations on Apple M3
  (8\,GB unified memory, 8 CPU cores). Log scale. Metal uses a stack-allocated
  scratchless shader at $n=49$, a single-pass uint32 index buffer for $n \leq 5{,}000$,
  and a batched fallback for $n=10{,}000$.
  Swift parallel (not shown) achieves 3.1M, 349k, 88k, 34k, 17k perm/s respectively.
}
\label{fig:throughput}
\end{figure}

% Figure 2: Speedup over NumPy (line chart)
% ---------------------------------------------------------------------------

\begin{figure}[ht]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=0.88\textwidth,
  height=6.5cm,
  xlabel={Dataset size $n$},
  ylabel={Speedup over NumPy},
  xlabel style={font=\small},
  ylabel style={font=\small},
  xticklabel style={font=\small},
  yticklabel style={font=\small},
  xtick={0,1,2,3,4},
  xticklabels={49, 500, 2{,}000, 5{,}000, 10{,}000},
  ymin=0, ymax=85,
  ytick={0,10,20,30,40,50,60,70,80},
  grid=major,
  grid style={dotted, gray!40},
  legend style={
    at={(0.98,0.98)},
    anchor=north east,
    font=\small,
    draw=gray!50,
    cells={anchor=west},
  },
  title style={font=\small\bfseries},
  title={Speedup over NumPy baseline},
]

% Numba dashed line
\addplot[
  color=clteal,
  mark=square*,
  mark size=2.5pt,
  line width=1.2pt,
  dashed,
] table[x expr=\coordindex, y=numba] {\speedupdata};
\addlegendentry{Numba parallel}

% Metal solid line
\addplot[
  color=clblue,
  mark=*,
  mark size=2.5pt,
  line width=1.2pt,
] table[x expr=\coordindex, y=metal] {\speedupdata};
\addlegendentry{Swift+Metal}

% Reference line at 1x
\addplot[
  color=clgray,
  line width=0.8pt,
  dotted,
  domain=0:4,
] {1};
\addlegendentry{NumPy (1$\times$)}

% Annotate the peak
\node[font=\tiny, clblue, anchor=south west]
  at (axis cs: 0, 77.8) {77.8$\times$};

\end{axis}
\end{tikzpicture}
\caption{
  Speedup relative to NumPy baseline (99,999 permutations).
  Swift+Metal achieves 77.8$\times$ at $n=49$ using the scratchless GPU shader,
  converging toward NumPy performance as $n$ grows and batch dispatch overhead
  dominates. Numba parallel provides consistent but modest speedup across all sizes.
  Swift CPU parallel (not shown) achieves 16.2$\times$ at $n=49$, declining to
  $\approx3\times$ at $n \geq 500$ — consistently outperforming Numba at scale.
}
\label{fig:speedup}
\end{figure}
